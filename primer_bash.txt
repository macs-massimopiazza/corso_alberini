clear => Pulisce il terminale
cd => Cambia cartella. Senza parametri torna alla home
cd .. => Vai alla cartella genitore di quella attuale (la cartella indietro)
pwd => Stampa cartella corrente, aka quella in cui ci si trova
ls => Mostra il contenuto di una cartella. Senza parametri mostra contenuto cartella corrente
ls -l => Mostra il contenuto "esteso" si una cartella
ls <path_cartella> => Mostra il contenuto di <path_cartella>
mkdir <nome_cartella> => Crea la cartella <nome_cartella> nella directory corrente
mv <a> <b> => Sposta (o rinonima) <a> in <b>
cp <a <b> => Copia <a> in <b>. <a> e <b> devono essere file
cp -r <a> <b> => Copia <a> in <b>. <a> può essere una cartella
rm <a> => Cancella <a>. <a> deve essere un file
rm -r <a> => Cancella <a>. <a> può essere una cartella
touch <nome_file> => Crea un file vuoto <nome_file>
cat <nome_file> => Stampa il contenuto di <nome_file>
nano <nome_file> => Permette di scrivere dentroo un file da terminale
echo <blabla> => Scrive <blabla> sul terminale
echo "<blabla>" > <nome_file> => Sovrascrive <nome_file> con <blabla>
echo "<blabla>" >> <nome_file> => Scrive in fondo a <nome_file> <blabla>
exit => Chiude il terminale.
head -n<numero-righe> <nome_file> => Stampa le prime <numero_righe> di <nome_file>
tail -n<numero-righe> <nome_file> => Stampa dal basso le ultime  <numero_righe> di <nome_file>
